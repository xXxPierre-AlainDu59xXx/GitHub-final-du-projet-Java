3   import java.awt.Image;
4   import java.io.IOException;
5   
6   import javax.imageio.ImageIO;
7   
8   import model.IAnimateSprite;
9   
10  /**
11  *<b>The AnimateSprite class animates pictures</b>
12  *The AnimateSprite class animates pictures
13  * The class AnimateSprite consists to make a movement for a picture
14  * @author Corentin Devrouete
15  * @version 1.1
16  */
17  public class AnimateSprite extends Sprite implements IAnimateSprite{
18  
19  	public Image[] images;
20  	public int next;
21  	/**
22  	 * 
23  	 * @param url
24  	 *  the default image link
25  	 * @param urls
26  	 *  this is the set of image links
27  	 */
28  	public AnimateSprite(String url,String[] urls) {
29  		super(url);
30  		this.images = new Image[urls.length];
31  		for (int a=0; a<images.length; a++ ){
32  			try {
33  				this.images[a] = ImageIO.read(this.getClass().getClassLoader().getResourceAsStream(urls[a]));
34  			} catch (final IOException e) {
35  				e.printStackTrace();
36  			}
37  		}this.next = 0;
38  	}
39  	
40  	/* (non-Javadoc)
41  	 * @see model.IAnimateSprite#nextImage()
42  	 */
43  	public void nextImage(){
44  		this.next ++;
45  		if (this.next>= this.images.length){
46  			this.next = 0;
47  		}
48  	}
49  	/*
50  	 * (non-Javadoc)
51  	 * @see model.Sprite#getImage()
52  	 */
53  	public Image getImage() {
54          return this.images[this.next];
55      }
56  }
