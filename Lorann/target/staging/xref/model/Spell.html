9   /**
10  *<b>The Spell class represents the sorcery that the player can launch</b>
11  * The class Spell consists to instantiate an object Spell with:
12  * <ul>
13  * <li>An x and y coordinates to locate on a map the spell.</li>
14  * <li>A url to connect the spell to an image format . png.</li>
15  * <li>A permeability</li>

23  	/**

26  	 * 			The x position of the spell on a landmark o,x,y.
27  	 * @param y
28  	 * 			The y position of the spell on a landmark o,x,y.
29  	 * @param url
30  	 * 			The link to a specific image. 
31  	 */
32  	public Spell ( int x, int y, String url){
33  		super(url,Permeability.PENETRABLE);

35  		setY(y);
36  		String images[] = {

38                  "sprite/fireball_2.png",
39                  "sprite/fireball_3.png",
40                  "sprite/fireball_4.png"};
41          this.sprite = new AnimateSprite(url,images);
42  	}
43

46  	 */

49  	}
