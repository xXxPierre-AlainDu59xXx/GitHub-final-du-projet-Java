3   import model.ILorann;
4   import controller.Permeability;
5   import model.SpellState;
6   import model.LorannState;
7   /**
8   *<b>The Lorann class represents the main character that the user plays</b>
9   * The class Lorann consists to instantiate an object Lorann with:
10  * <ul>
11  * <li>An x and y coordinates to locate on a map at the start of the game.</li>
12  * <li>A url to connect the character to an image format . png.</li>
13  * <li>A permeability</li>
14  * </ul>
15  * @author Corentin Devrouete
16  * @version 1.4
17  */
18  public class Lorann extends MobileElement implements ILorann {
19  	
20  	public LorannState state = LorannState.DOWN;
21  	public SpellState spellState;
22  	
23  	
24  	/**
25  	 * Initialize Lorann and characteristics of Lorann 
26  	 * @param x
27  	 * 			The x position of the character on a landmark o,x,y.
28  	 * 			It can be changed by the user during the game.
29  	 * @param y
30  	 * 			The y position of the character on a landmark o,x,y.
31  	 * 			It can be changed by the user during the game.
32  	 * @param url
33  	 * 			The link to a specific image. 
34  	 */
35  	public Lorann (int x, int y, String url){
36  		super(url,Permeability.PENETRABLE);
37  		setX(x);
38  		setY(y);
39  		spellState = SpellState.NotThrow;
40  		String images[] = {
41  				"sprite/lorann_b.png",
42                  "sprite/lorann_br.png",
43                  "sprite/lorann_r.png",
44                  "sprite/lorann_ur.png",
45                  "sprite/lorann_u.png",
46                  "sprite/lorann_ul.png",
47                  "sprite/lorann_l.png",
48                  "sprite/lorann_bl.png"};
49          this.sprite = new AnimateSprite(url,images);
50  	}
51  
52  	
53  	/*
54  	 * (non-Javadoc)
55  	 * @see contract.ILorann#getState()
56  	 */
57  	public LorannState getState() {
58  		return state;
59  	}
60  
61  	/*
62  	 * (non-Javadoc)
63  	 * @see contract.ILorann#setState(contract.LorannState)
64  	 */
65  	public void setState(LorannState state) {
66  		this.state = state;
67  	}
68  
69  	/*
70  	 * (non-Javadoc)
71  	 * @see contract.ILorann#getPossessionSpell()
72  	 */
73  	public SpellState getPossessionSpell() {
74  		return spellState;
75  	}
76  
77  	/*
78  	 * (non-Javadoc)
79  	 * @see contract.ILorann#setPossessionSpell(contract.SpellState)
80  	 */
81  	public void setPossessionSpell(SpellState spellState) {
82  		this.spellState = spellState;
83  	}
84  }
